52## @brief Rotates the drone path about the Z axis
53# @details Used to correct when the mat is not aligned with global lighthouse coordinates
54# @param sequence List of position tuples (x, y, z) to rotate
55# @param theta Rotation angle in radians
56# @return List of rotated position tuples
57def rotate_sequence(sequence, theta):
58 rotated_sequence = []
59 cos_theta = math.cos(theta)
60 sin_theta = math.sin(theta)
61 for pos in sequence:
62 x = pos[0]
63 y = pos[1]
64 z = pos[2]
65 new_x = x * cos_theta - y * sin_theta
66 new_y = x * sin_theta + y * cos_theta
67 rotated_sequence.append((new_x, new_y, z))
68 return rotated_sequence

70## @brief Reads path waypoints from MATLAB CSV and returns a simplified sequence
71# @details Sequence contains only turning points and start/end points.
72# Can only detect new obstacles at corners and will miss obstacles in straight paths.
73# @param csv_file Path to the CSV file containing waypoint data
74# @return List of normalized waypoint tuples (x, y, z)
75def read_path_waypoints(csv_file='path.csv'):
76 points = []
77 try:
78 with open(csv_file, 'r') as file:
79 reader = csv.reader(file)
80 for row in reader:
81 if len(row) >= 2:
82 x, y = float(row[0]), float(row[1])
83 points.append([x, y])
84 except FileNotFoundError:
85 print(f"CSV file {csv_file} not found")
86 return []
87
88 if len(points) < 2:
89 return points
90
91 waypoints = [points[0]]
92
93 for i in range(1, len(points) - 1):
94 prev_dir = [points[i][0] - points[i-1][0], points[i][1] - points[i-1][1]]
95 next_dir = [points[i+1][0] - points[i][0], points[i+1][1] - points[i][1]]
96
97 if prev_dir != next_dir:
98 waypoints.append(points[i])
99
100 waypoints.append(points[-1])
101
102 if waypoints:
103 global offset_x, offset_y
104 offset_x, offset_y = waypoints[0][0], waypoints[0][1]
105 normalized_waypoints = []
106 for point in waypoints:
107 # Convert to tuples with height 0.4
108 normalized_waypoints.append(((point[0] - offset_x)/10, -(point[1] - offset_y)/10, 0.15))
109 return normalized_waypoints
110
111 return waypoints

197## @brief Flies the drone through a given sequence of positions
198# @details Checks for obstacles at each position and logs them if detected.
199# Sets starting point as (0,0,0) and moves relative to that point instead of global lighthouse coordinates.
200# Takes approximately 0.5 seconds to move from point to point.
201# @param scf SyncCrazyflie object

208 cf = scf.cf

212 cf.platform.send_arming_request(True)
213 time.sleep(1.0)
214

216 print('Setting position {}'.format(position))
217
218 x = position[0] + base_x
219 y = position[1] + base_y
220 z = position[2] + base_z
221
222 for j in range(10):
223 cf.commander.send_position_setpoint(x, y, z, yaw)
224 time.sleep(0.1)

226 print(f"Reached position {position}")

228 ## @brief Check for obstacle over multiple readings to reduce false positives
229 black = 0
230 for j in range(50):
231 if over_obstacle():
232 black = black + 1
233 time.sleep(0.01)
234
235
236 print(f"Black count: {black}")
237 if black >= 40 and i != 0:
238 for j in range(30):
239 cf.commander.send_position_setpoint(initial_position[0], initial_position[1], 0.15, yaw)
240 time.sleep(0.1)

257## @brief Logging callback function that updates estimated position and shutter reading

282## @brief Main program entry point
283# @details Main program logic:
284# 1. Sets up logging
285# 2. Sets up initial position
286# 3. Reads path from MATLAB CSV (chooses waypoints or all blocks)
287# 4. Rotates path if needed
288# 5. Runs the sequence
289# 6. Prompts user to run again or exit
290if __name__ == '__main__':
291 cflib.crtp.init_drivers()
292
293
294 initial_yaw = 90 # In degrees
295
296 logconf = LogConfig(name='pos', period_in_ms=10)
297 logconf.add_variable('stateEstimate.x', 'float')
298 logconf.add_variable('stateEstimate.y', 'float')
299 logconf.add_variable('stateEstimate.z', 'float')
300 logconf.add_variable('motion.shutter', 'uint16_t')
301
302 while True:
303 print("Initial position: ", estimated_position[0], estimated_position[1], estimated_position[1])
304 with SyncCrazyflie(uri, cf=Crazyflie(rw_cache='./cache')) as scf:

307 time.sleep(0.1)

309 reset_estimator(scf)
310

312 sequence = read_path_waypoints('path.csv')
313 else:

315
316 rotate_sequence(sequence, math.radians(0))
317 run_sequence(scf, sequence, estimated_position[0], estimated_position[1], estimated_position[2], initial_yaw)
318
319 while True:
